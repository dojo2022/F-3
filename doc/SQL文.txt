user_id int PRIMARY KEY AUTO_INCREMENT NOT NULL,

user_pass VARCHAR(20) NOT NULL,

date_id int PRIMARY KEY AUTO_INCREMENT NOT NULL,

);

mission_id int PRIMARY KEY AUTO_INCREMENT NOT NULL,

);

CREATE TABLE MISSIONCLEAR (

sentence VARCHAR(30),

);
